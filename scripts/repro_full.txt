# Full reproduction script: boot -> title -> stage select -> enter stage
# Takes screenshots at each milestone for comparison

# Boot sequence - wait for title screen
TURBO ON
WAIT 400
SCREENSHOT C:/temp/repro_01_title.png
LOG === Title screen reached ===

# Press START to go to stage select
HOLD START
WAIT 3
RELEASE START
WAIT 200
SCREENSHOT C:/temp/repro_02_stageselect.png
LOG === Stage select reached ===

# Wait a moment for portraits to fully load
WAIT 60
SCREENSHOT C:/temp/repro_03_stageselect_stable.png
LOG === Stage select stable ===

# Select a stage (default cursor position) - press START
HOLD START
WAIT 3
RELEASE START
WAIT 300
SCREENSHOT C:/temp/repro_04_stage_entry.png
LOG === Stage entry (cutscene or gameplay) ===

# Wait for gameplay to begin
WAIT 300
SCREENSHOT C:/temp/repro_05_gameplay.png
LOG === Gameplay ===

# Hold for TCP inspection
TURBO OFF
WAIT 600
SCREENSHOT C:/temp/repro_06_gameplay_late.png
LOG === Late gameplay ===

EXIT 0
